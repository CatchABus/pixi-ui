Class ScrollBox

ScrollBox for arranging and scrolling pixi containers withing some area with scrolling

example
new ScrollBox({
    background: 0XFFFFFF,
    type: 'vertical',
    elementsMargin: 10,
    width: 200,
    height: 300,
    radius: 10,
    padding: 10,
    items: [
        new Graphics().beginFill(0x000000).drawRect(0, 0, 200, 50),
        new Graphics().beginFill(0x000000).drawRect(0, 0, 200, 50),
        new Graphics().beginFill(0x000000).drawRect(0, 0, 200, 50),
        new Graphics().beginFill(0x000000).drawRect(0, 0, 200, 50),
        new Graphics().beginFill(0x000000).drawRect(0, 0, 200, 50),
        new Graphics().beginFill(0x000000).drawRect(0, 0, 200, 50),
        new Graphics().beginFill(0x000000).drawRect(0, 0, 200, 50),
    ],
});
